local SilentAim = AimBot:addToggle("Silent Aim", function(value7)
	if value7 then
		-- veriables --
		local Plr = game:GetService("Players").LocalPlayer
		local mouse = Plr:GetMouse()
		local mousedown = false
		local isonmovething = false
		local target

		-- Shoot Through Walls --
		local mt = getrawmetatable(game)
		local nc = mt.__namecall

		setreadonly(mt, false)

		mt.__namecall = newcclosure(function(self, ...)
			local Method = getnamecallmethod()
			local Arguments = {...}

			if Method == 'FindPartOnRayWithIgnoreList' and value7 then
				table.insert(Arguments[2], game.GetService(game, 'Workspace').Map)

				return nc(self, unpack(Arguments))
			end

			return nc(self,...)
		end)

		setreadonly(mt, true)

		-- mouse event for the Silent Aim

		mouse.Button1Down:connect(function()
			if game.Players.LocalPlayer:FindFirstChild("__SPAWNED")then
				target=nil
				for _,v in pairs(game.Players:GetChildren())do
					if 
						v.Character and
						v.Character.Parent==game.Workspace and
						(
							v.Team~=game.Players.LocalPlayer.Team or
								(
									game:GetService("Workspace")["__VARIABLES"].RoundType.Value=="FFA"and
									v~=game.Players.LocalPlayer and
									math.random(1,0)
								)
						)and 
							v.Character:FindFirstChild("Humanoid")and
							v:FindFirstChild("__SPAWNED")and
							not v.Character:FindFirstChild("ForceField")then
						target=v.Character
						require(game:GetService("ReplicatedStorage").Framework.Modules["2 | Network"]).Fire("New Projectile",1,1,math.floor(game.Workspace.DistributedGameTime))
						break
					end
				end
			end
		end)

		-- hook namecall,  detect projectile and fire the do damage remote

		local namecall
		namecall=hookmetamethod(game,"__namecall",function(a,b,c,d,e,...)
			if tostring(getnamecallmethod())=="FireServer"and tostring(a)=="new projectile"and target~=nil then
				game.Workspace.__THINGS.__REMOTES["do damage"]:FireServer(
					{
						target.Humanoid,
						b[2],
						235,
						target.HumanoidRootPart.Position,
						false
					}
				)
			end
			return namecall(a,b,c,d,e,...)
		end)
	end
end)
